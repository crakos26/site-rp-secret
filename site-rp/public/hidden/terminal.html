<!DOCTYPE html>

<html>

<body style="background:black;color:#0f0;font-family:monospace;">


<div id="term"></div>

<input id="input" autofocus>


<script>

let term = document.getElementById("term");


function print(t) {

  term.innerHTML += t + "<br>";

}


print(">> SECURE TERMINAL");

print(">> type help");


document.getElementById("input").addEventListener("keydown", (e) => {

  if (e.key === "Enter") {

    let cmd = e.target.value;

    e.target.value = "";


    if (cmd === "help") {

      print("commands: logs, mission, sheets, clear");

    }

    else if (cmd === "logs") {

      print("[LOG] accès sécurisé");

      print("[LOG] utilisateur connecté");

    }

    else if (cmd === "mission") {

      print("Mission: surveiller la zone portuaire");

    }

    else if (cmd === "sheets") {

      window.location.href = "sheets.html";

    }

    else if (cmd === "clear") {

      term.innerHTML = "";

    }

    else {

      print("commande inconnue");

    }

  }

});

</script>


</body>

</html>
